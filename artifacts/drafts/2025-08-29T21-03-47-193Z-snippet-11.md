def plan(goal, g, priors, budget_ms):
    candidates = g.paths(from_=goal, via_kind="proc", to_kind="actuator")
    scored = [(p, score_path(p, priors, budget_ms)) for p in candidates]
    return max(scored, key=lambda x:x[1])[0]
